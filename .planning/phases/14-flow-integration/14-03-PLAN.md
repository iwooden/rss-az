        gs.initialize_game(seed=200)

        corp = CORPS[CORP_NAMES[0]]  # BME
        corp.set_active(gs, True)
        corp.set_cash(gs, 50)  # Enough for face value
        corp.set_in_receivership(gs, True)  # No president

        # FI owns company 0 (face value typically 5-10)
        from entities.fi import FI
        from entities.company import COMPANIES
        company = COMPANIES[0]
        company.transfer_to_fi(gs)

        face_value = company.get_face_value()
        initial_corp_cash = corp.get_cash(gs)
        initial_fi_cash = FI.get_cash(gs)

        # Trigger offer generation and presentation
        from phases.acquisition import setup_acquisition_phase_py
        setup_acquisition_phase_py(gs)

        # If receivership corp was offered FI company, it should auto-buy
        # Check: corp lost money, FI gained money, company in acquisition zone
        if corp.has_acquisition_company(gs, 0):
            # Auto-buy happened
            assert corp.get_cash(gs) == initial_corp_cash - face_value
            assert FI.get_cash(gs) == initial_fi_cash + face_value
            # No active offer (was processed automatically)
            # acq_active_corp should be -1 or pointing to next offer
        # If no acquisition, means no FI offer was generated for this corp

    def test_receivership_skips_unaffordable_fi(self):
        """Receivership corp auto-passes on FI offer when cannot afford."""
        gs = GameState()
        gs.initialize_game(seed=201)

        # Setup: receivership corp with NO cash
        corp = CORPS[CORP_NAMES[0]]
        corp.set_active(gs, True)
        corp.set_cash(gs, 0)  # Can't afford anything
        corp.set_in_receivership(gs, True)

        # FI owns company
        from entities.fi import FI
        from entities.company import COMPANIES
        company = COMPANIES[0]
        company.transfer_to_fi(gs)

        initial_fi_cash = FI.get_cash(gs)

        # Trigger
        from phases.acquisition import setup_acquisition_phase_py
        setup_acquisition_phase_py(gs)

        # Corp should have auto-passed, no acquisition
        assert not corp.has_acquisition_company(gs, 0)
        # FI cash unchanged
        assert FI.get_cash(gs) == initial_fi_cash
