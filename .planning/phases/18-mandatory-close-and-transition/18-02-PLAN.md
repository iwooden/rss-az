   class TestMandatoryClose:

       def test_mandatory_close_not_triggered_positive_total(self, game_state):
           """Mandatory close does nothing when income + cash >= 0."""
           # Player has $30 cash, no companies -> income 0
           # Total = $30 >= 0, no close needed

       def test_mandatory_close_triggered_negative_total(self, game_state):
           """CLO-14: Mandatory close triggers when income + cash < 0."""
           # Set up: player with negative income that exceeds cash
           # Set CoO level high
           TURN.set_coo_level(game_state, 8)

           # Give player a 1-star company with negative adjusted income
           # Company 0: $1 income, 1 star, CoO 8 = $8 -> adjusted = -$7
           PLAYERS[0].set_owns_company(game_state, 0, True)
           COMPANIES[0].set_location(game_state, 3, 0)  # LOC_PLAYER

           # Reduce player cash to trigger mandatory close
           # Cash = $5, income = -$7, total = -$2 < 0
           PLAYERS[0].set_cash(game_state, 5)

           process_mandatory_close_py(game_state)

           # Company should be closed
           assert not PLAYERS[0].owns_company(game_state, 0)
           assert COMPANIES[0].is_removed(game_state)

       def test_mandatory_close_cheapest_first(self, game_state):
           """CLO-15: Cheapest (lowest face value) negative-income company closed first."""
           TURN.set_coo_level(game_state, 8)

           # Give player two negative-income companies with different face values
           # Company 0: face value $1, 1 star (cheapest)
           # Company 10: face value $5, 2 stars (more expensive)
           PLAYERS[0].set_owns_company(game_state, 0, True)
           COMPANIES[0].set_location(game_state, 3, 0)
           PLAYERS[0].set_owns_company(game_state, 10, True)
           COMPANIES[10].set_location(game_state, 3, 0)

           # Set cash so closing ONE company makes total >= 0
           # income = (-7) + (-12) = -19 at CoO 8
           # If cash = $10, total = -9, need to close at least one
           # After closing company 0 (-7), income = -12, total = -12 + 10 = -2, still negative
           # After closing company 10 (-12), income = 0, total = 0 + 10 = 10 >= 0
           # But requirement is close CHEAPEST first (company 0)
           # So with cash = $20: total = -19 + 20 = 1 >= 0, no close needed
           # With cash = $18: total = -19 + 18 = -1 < 0, close cheapest (company 0)
           # After close 0: income = -12, total = -12 + 18 = 6 >= 0, stop

           PLAYERS[0].set_cash(game_state, 18)
           # Income at CoO 8: company 0 = 1-8 = -7, company 10 = 4-16 = -12
           # Total income = -19, total = -19 + 18 = -1 < 0

           process_mandatory_close_py(game_state)

           # Cheapest (company 0, face value $1) should be closed
           assert not PLAYERS[0].owns_company(game_state, 0)
           assert COMPANIES[0].is_removed(game_state)

           # More expensive company 10 should still be owned
           assert PLAYERS[0].owns_company(game_state, 10)
           assert not COMPANIES[10].is_removed(game_state)

   class TestClosingPhaseTransition:
       """Tests for CLOSING phase transition (CLO-16)."""

       def test_phase_transitions_after_mandatory_close(self, game_state):
           """CLO-16: Phase transitions to INCOME (INVEST) when offers processed and mandatory close complete."""
           from core.driver import GameDriver
           from core.data import GamePhases

           driver = GameDriver()

           # Advance to CLOSING phase
           # Start in INVEST, pass all players to reach WRAP_UP
           game_state.set_phase(GamePhases.PHASE_INVEST)
           game_state._set_active_player(0)

           # Simulate state where CLOSING has no offers and mandatory close has nothing to do
           # All players have positive income + cash
           game_state.set_phase(GamePhases.PHASE_CLOSING)

           # Process until phase changes (auto-close + offer generation + mandatory close + transition)
           # Use step with history tracking
           history = []
           max_iterations = 100
           for _ in range(max_iterations):
               driver.step(game_state, history=history)
               current_phase = game_state.get_phase()
               if current_phase != GamePhases.PHASE_CLOSING:
                   break

           # Should have transitioned to INVEST (temporary target for INCOME)
           assert game_state.get_phase() == GamePhases.PHASE_INVEST

       def test_closing_full_flow_with_mandatory_close(self, game_state):
           """Integration test: CLOSING with offers processed then mandatory close triggers."""
           from core.driver import GameDriver
           from core.data import GamePhases
           from phases.closing import (
               apply_closing_auto_py, get_close_offer_count_py
           )

           # Set up: high CoO to create negative-income companies
           TURN.set_coo_level(game_state, 8)

           # Give player 0 a negative-income company
           PLAYERS[0].set_owns_company(game_state, 0, True)
           COMPANIES[0].set_location(game_state, 3, 0)

           # Low cash to trigger mandatory close after offer processing
           PLAYERS[0].set_cash(game_state, 5)

           # Enter CLOSING phase
           game_state.set_phase(GamePhases.PHASE_CLOSING)

           # Process auto-close and generate offers
           apply_closing_auto_py(game_state)

           # There should be close offer(s) generated
           offer_count = get_close_offer_count_py(game_state)

           driver = GameDriver()

           # Process offers (pass on all to trigger mandatory close)
           # Get legal actions and apply PASS
           from core.actions import ACTION_PASS
           max_iterations = 50
           for _ in range(max_iterations):
               current_phase = game_state.get_phase()
               if current_phase != GamePhases.PHASE_CLOSING:
                   break

               actions = driver.get_legal_actions(game_state)
               if len(actions) == 0:
                   # Non-player phase, step
                   driver.step(game_state)
                   continue

               # Find PASS action and apply it
               pass_action = None
               for action in actions:
                   if action[0] == ACTION_PASS:
                       pass_action = action
                       break

               if pass_action:
                   driver.apply_action(game_state, pass_action)
               else:
                   # Only CLOSE action available, apply it
                   driver.apply_action(game_state, actions[0])

           # Phase should have transitioned
           # If player passed on all offers, mandatory close should have closed the company
           # Either way, should now be in INVEST
           assert game_state.get_phase() == GamePhases.PHASE_INVEST

           # Company should be closed (either by offer accept or mandatory close)
           assert COMPANIES[0].is_removed(game_state) or not PLAYERS[0].owns_company(game_state, 0)
